    var body: some View {
        // Encase in a container that fills the entire NSPanel frame but aligns elements to the top.
        VStack(spacing: 0) {
            ZStack(alignment: .bottom) {
                // To blend perfectly with the hardware notch, it must be pure pitch black.
                LiquidNotchShape(bottomRadius: isExpanded ? 24 : notchCornerRadius)
                    .fill(Color.black)
                    .shadow(color: Color.black.opacity(0.3), radius: 8, x: 0, y: 4)
                    .zIndex(0)
                
                // Show contents ONLY when clicked/expanded
                if isExpanded {
                    VStack(spacing: 12) {
                        HStack(spacing: 12) {
                            // Album Artwork
                            if let cover = spotify.coverImage {
                                Image(nsImage: cover)
                                    .resizable()
                                    .aspectRatio(contentMode: .fill)
                                    .frame(width: 46, height: 46)
                                    .cornerRadius(10)
                                    .padding(.leading, 18)
                            } else {
                                Image(systemName: "music.note.circle.fill")
                                    .resizable()
                                    .aspectRatio(contentMode: .fit)
                                    .frame(width: 40, height: 40)
                                    .foregroundColor(.green)
                                    .padding(.leading, 18)
                            }
                            
                            // Text Meta Data
                            VStack(alignment: .leading, spacing: 2) {
                                Text(spotify.trackName)
                                    .foregroundColor(.white)
                                    .font(.system(size: 14, weight: .bold))
                                    .lineLimit(1)
                                
                                Text(spotify.artistName.isEmpty ? "Spotify" : spotify.artistName)
                                    .foregroundColor(.white.opacity(0.7))
                                    .font(.system(size: 12, weight: .medium))
                                    .lineLimit(1)
                            }
                            
                            Spacer()
                            
                            // Mac Icon to toggle volume
                            Button(action: {
                                withAnimation(.spring(response: 0.4, dampingFraction: 0.6)) {
                                    isVolumeExpanded.toggle()
                                }
                            }) {
                                Image(systemName: "applelogo")
                                    .font(.system(size: 18))
                                    .foregroundColor(.white.opacity(isVolumeExpanded ? 1.0 : 0.6))
                            }.buttonStyle(.plain)
                            
                            // Dynamic Waveform "Sound beat"
                            WaveformView(isPlaying: spotify.isPlaying)
                                .padding(.trailing, 22)
                        }
                        
                        // Volume Slider
                        if isVolumeExpanded {
                            HStack {
                                Image(systemName: "speaker.fill").foregroundColor(.white.opacity(0.6))
                                Slider(value: Binding(
                                    get: { spotify.volume },
                                    set: { v in spotify.volume = v; spotify.setVolume(v) }
                                ), in: 0...100)
                                .tint(.white)
                                Image(systemName: "speaker.wave.3.fill").foregroundColor(.white.opacity(0.6))
                            }
                            .padding(.horizontal, 22)
                            .transition(.opacity.combined(with: .move(edge: .top)))
                        }
                        
                        // Progress / Loadbar
                        HStack(spacing: 8) {
                            Text(formatTime(spotify.position))
                                .font(.system(size: 10, weight: .medium, design: .monospaced))
                                .foregroundColor(.gray)
                            
                            Slider(value: Binding(
                                get: { isEditingPosition ? editingPosition : spotify.position },
                                set: { v in editingPosition = v }
                            ), in: 0...max(spotify.duration, 1)) { editing in
                                isEditingPosition = editing
                                if !editing { spotify.seek(to: editingPosition) }
                            }
                            .tint(.white)
                            
                            Text(formatTime(spotify.duration))
                                .font(.system(size: 10, weight: .medium, design: .monospaced))
                                .foregroundColor(.gray)
                        }
                        .padding(.horizontal, 22)
                        
                        // Media Controls
                        HStack(spacing: 30) {
                            Button(action: { spotify.toggleShuffle() }) {
                                Image(systemName: "shuffle")
                                    .font(.system(size: 16))
                                    .foregroundColor(spotify.isShuffling ? .green : .white.opacity(0.6))
                            }.buttonStyle(.plain)
                            
                            Button(action: { spotify.previousTrack() }) {
                                Image(systemName: "backward.fill")
                                    .font(.system(size: 20))
                                    .foregroundColor(.white)
                            }.buttonStyle(.plain)
                            
                            Button(action: { spotify.playPause() }) {
                                Image(systemName: spotify.isPlaying ? "pause.fill" : "play.fill")
                                    .font(.system(size: 28))
                                    .foregroundColor(.white)
                            }.buttonStyle(.plain)
                            
                            Button(action: { spotify.nextTrack() }) {
                                Image(systemName: "forward.fill")
                                    .font(.system(size: 20))
                                    .foregroundColor(.white)
                            }.buttonStyle(.plain)
                        }
                        .padding(.bottom, 16)
                    }
                    .padding(.top, 16)
                    .zIndex(1)
                    .transition(
                        .asymmetric(
                            insertion: .opacity.animation(.easeIn(duration: 0.2).delay(0.1)),
                            removal: .opacity.animation(.easeOut(duration: 0.02))
                        )
                    )
                }
            }
            .contentShape(Rectangle())
            // Dynamic heights for expansion
            .frame(width: isExpanded ? 360 : baseNotchWidth,
                   height: isExpanded ? (isVolumeExpanded ? 180 : 145) : dynamicNotchHeight)
            .scaleEffect((isHovering && !isExpanded) ? 1.03 : 1.0, anchor: .top)
            .onTapGesture {
                withAnimation(.spring(response: 0.4, dampingFraction: 0.6, blendDuration: 0.2)) {
                    isExpanded.toggle()
                }
            }
            
            // Provides invisible empty space to fill the rest of the NSPanel so layout remains stable
            Spacer()
                .allowsHitTesting(false)
        }
        .frame(maxWidth: .infinity, maxHeight: .infinity)
        .onAppear {
            detectNotchMetrics()
        }
        .onReceive(timer) { _ in
